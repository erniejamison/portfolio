Foobar

This ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 

n = 0
print "How many items would you like to see? "
i = gets.chomp.to_i

def foobar(i)
  n = 1
  while n <= i
    remainder_when_divided_by_3 = n % 3
    remainder_when_divided_by_5 = n % 5

    if remainder_when_divided_by_3 == 0 && remainder_when_divided_by_5 == 0
      puts "FooBar"
    elsif remainder_when_divided_by_3 == 0
      puts "Foo"
    elsif remainder_when_divided_by_5 == 0
      puts "Bar"
    else remainder_when_divided_by_3 != 0 && remainder_when_divided_by_5 != 0
      puts n
    end
    n = n + 1
  end
end

foobar(i)
puts "Goodbye"
